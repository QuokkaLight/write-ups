```python
 -*- coding: utf-8 -*-

import codecs

c = codecs.open('README.txt','r','utf-8').read()

symbols = { u'♀' : 2, u'⊕' : 3, u'♆' : 8, u'♇' : 9, ' ' : '' }
d = [ str(symbols[i]) for i in c ]


print ''.join(d)
```
